125 // push an item to the queue.
126 template<typename T>
127 void QueueList<T>::push (const T i) {
128  // create a temporary pointer to tail.
129  link t = tail;
130 
131  // create a new node for the tail.
132  tail = (link) new node;
133 
134  // if there is a memory allocation error.
135  if (tail == NULL)
136  exit ("QUEUE: insufficient memory to create a new node.");
137 
138  // set the next of the new node.
139  tail->next = NULL;
140 
141  // store the item to the new node.
142  tail->item = i;
143 
144  // check if the queue is empty.
145  if (isEmpty ())
146  // make the new node the head of the list.
147  head = tail;
148  else
149  // make the new node the tail of the list.
150  t->next = tail;
151 
152  // increase the items.
153  size++;
154 }
